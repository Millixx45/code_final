<script>
    const QUESTIONS = [
        { q: "Wurde deine Wahrnehmung oft infrage gestellt? ('Das hast du dir eingebildet')", cat: 0 },
        { q: "Gab es einen extremen Wechsel zwischen Idealisierung und Entwertung?", cat: 0 },
        { q: "Hat sie kontrolliert, mit wem du Zeit verbringst?", cat: 1 },
        { q: "Musstest du ständig erreichbar sein, um Streit zu vermeiden?", cat: 1 },
        { q: "Wurden deine Erfolge systematisch kleingeredet?", cat: 2 },
        { q: "Hat sie dich vor anderen Frauen herabgesetzt?", cat: 2 },
        { q: "Waren ihre emotionalen Reaktionen völlig unvorhersehbar?", cat: 3 },
        { q: "Gab es ein Muster von impulsiven Trennungen und Versöhnungen?", cat: 3 },
        { q: "Hat sie jegliche Verantwortung für Fehler abgelehnt?", cat: 4 },
        { q: "Fühltest du dich in der Beziehung emotional verhungert?", cat: 4 }
    ];

    let current = 0;
    let scores = new Array(10).fill(0);
    let chart;

    function init() {
        const container = document.getElementById('test-container');
        QUESTIONS.forEach((item, i) => {
            const div = document.createElement('div');
            div.className = `question-step ${i === 0 ? 'active' : ''}`;
            div.id = `q-${i}`;
            div.innerHTML = `
                <h3 style="font-family: Montserrat; font-size: 1.3rem; min-height: 80px;">${item.q}</h3>
                <div class="points-grid">${[0,1,2,3,4,5].map(p => `<button class="point-btn" onclick="next(${i}, ${p})">${p}<span class="point-label">${p==0?'Nie':p==5?'Immer':''}</span></button>`).join('')}</div>
            `;
            container.appendChild(div);
        });

        chart = new Chart(document.getElementById('liveChart'), {
            type: 'radar',
            data: { labels: ['Manipulation', 'Kontrolle', 'Abwertung', 'Chaos', 'Egoismus'], datasets: [{ data: [0,0,0,0,0], backgroundColor: 'rgba(196, 131, 110, 0.25)', borderColor: '#C4836E', borderWidth: 2 }] },
            options: { scales: { r: { min: 0, max: 10, ticks: { display: false }, grid: { color: 'rgba(255,255,255,0.1)' } } }, plugins: { legend: { display: false } } }
        });
    }

    function next(idx, p) {
        scores[idx] = p;
        const catS = [0,0,0,0,0];
        QUESTIONS.forEach((q, i) => { catS[q.cat] += scores[i]; });
        chart.data.datasets[0].data = catS; chart.update();
        document.getElementById('score-val').textContent = scores.reduce((a,b)=>a+b, 0);

        if (current < 9) {
            document.getElementById(`q-${current}`).classList.remove('active');
            current++;
            document.getElementById(`q-${current}`).classList.add('active');
            document.getElementById('prog-fill').style.width = ((current+1)*10) + "%";
            document.getElementById('prog-text').textContent = `FRAGE ${current+1} VON 10`;
        } else {
            finish();
        }
    }

    function finish() {
        document.getElementById('modal').classList.add('active');
        document.getElementById('res-desc').innerHTML = "Deine Ergebnisse deuten auf ein starkes Trauma Bonding hin. Dies erklärt, warum das 'Ex-Freundin loslassen' momentan so schmerzhaft ist.";
    }

    window.onload = init;
</script>
